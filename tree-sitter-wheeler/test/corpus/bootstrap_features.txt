    region arena = new region(44, 3);
    words data = allocate(arena, 2);
    set(data, 0, 9);
    result = data[0];
    bytes raw = allocateBytes(arena, 4);
    setByte(raw, 0, 65);
    boolean valid = utf8Valid(raw);
    long scalarCount = utf8Count(raw);
    long byteCount = bufferLength(raw);
    long firstScalar = utf8Scalar(raw, 0);
    long firstWidth = utf8Width(raw, 0);
    utf8 text = freezeUtf8(raw);
    long textScalars = utf8Count(text);
    longmap symbols = allocateMap(arena, 1);
    put(symbols, 1, 9);
    long symbol = mapGet(symbols, 1);
    boolean present = mapHas(symbols, 1);
    drop(symbols);
    drop(text);
    drop(data);
    drop(arena);
  }
