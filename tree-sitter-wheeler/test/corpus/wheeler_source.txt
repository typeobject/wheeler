classical class Counter {

  entry void main() {
    increment();
    reverse { increment(); }
    assert count == 0;
  }

  entry void main() {

    bell();

  variant Option {
    case None();
    case Some(long value);
  }
  state long selected = 0;
  entry void main() {
    Option value = new Option.Some(9);
    match (value) {
      case Option.None() { selected = 0; }
      case Option.Some(long payload) { selected = payload; }
    }
  }
}
