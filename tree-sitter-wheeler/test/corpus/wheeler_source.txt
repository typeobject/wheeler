classical class Counter {
  state long count = 0;
  rev void increment() { count += 1; }
  entry void main() {
    increment();
    reverse { increment(); }
    assert count == 0;
  }
}

quantum class Bell {
  state long measured = 0;
  qreg q = new qreg(2);
  unitary void bell() {
    H(q[0]);
    CNOT(q[0], q[1]);
  }
  entry void main() {
    prepare(q, 0);
    bell();
    measured = measure(q);
  }
}
